\begin{frame}
    \frametitle{L'association Clusir-Est}
    \begin{center}
        \begin{itemize}
            \itemcol Une association loi 1901 à but non lucratif et non commercial.
            \newline
            \itemcol Le but est de développer toute action propre à une
            meilleure approche et maîtrise des risques du SI.
            \newline
            \itemcol Le Clusir-Est n'est pas une "sous-entité" du Clusif :
            L'association mène des actions "dans le respect du code
            d'éthique des métiers de la sécurité des SI défini par le Clusif".
            \newline
            \itemcol Les membres choisissent les thèmes à aborder (via échanges et/ou sondages)
        \end{itemize}
    \end{center}
\end{frame}

\begin{frame}
    \frametitle{Adhésion au Clusir-Est}
    Les modalités :
    \begin{center}
        \begin{itemize}
\itemcol L'association est à but non lucratif.
\itemcol Les seuls moyens financiers proviennent des cotisations versées par
ses adhérents
\itemcol Adhésion :
        \begin{itemize}
        \itemcol 100 euros annuellement (quelle que soit la période de l'année)
        \itemcol réduction de 50\% accordée pour le 2ème (et suivant) adhérent
        d'une société déjà adhérente
        \itemcol les membres du bureau payent leur cotisation
        \end{itemize}
        \end{itemize}
    \end{center}
